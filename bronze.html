    materialized='incremental',

WITH source AS (
  SELECT * FROM {{ source('raw', 'salic_artistas') }}
  WHERE load_date >= CURRENT_DATE - INTERVAL '1 day'

  SELECT
    regexp_replace(cpf, '[^0-9]', '') AS cpf,
    upper(trim(nome)) AS nome,
    to_date(data_cadastro, 'YYYY-MM-DD') AS data_cadastro,
    current_timestamp() AS load_timestamp,
    'salic' AS source_system
  FROM source
)
